                el.style.opacity = 0
            },
            enter: function (el, done) {
                Velocity(el, {opacity: 1}, {duration: 1000, complete: done})
            },
            afterEnter: function (el) {
                alert('动画结束')
            },
            beforeLeave: function (el) {
                el.style.opacity = 0
            },
            leave: function (el, done) {
                Velocity(el, {opacity: 1},{duration: 1000, complete: done})
            },
            afterLeave: function (el) {
                alert('动画结束')
            }
        }
    })

##### Vue中的动画封装

```
   <div id="app">
        <fade :show='show'>
            <h1>hello world</h1>
        </fade>
        <button @click='btnClick'>按扭</button>
    </div>
   Vue.component('fade', {
        props: ['show'],
        template: `<transition @before-enter='tranBeforeEnter'
                                @enter='tranEnter'>
                     <slot v-if="show"> </slot>
                   </transition>`,
        methods: {
            tranBeforeEnter: function (el) {
                el.style.color = 'red'
            },
            tranEnter: function (el, done) {
                setTimeout(() => {
                    el.style.color = 'green';
                    done()
                }, 2000)
            }
        }
    })

    var app = new Vue({
        el: '#app',
        data: {
            show: true
        },
        methods: {
            btnClick: function () {
                this.show = !this.show;
            }
        }
    })
```
